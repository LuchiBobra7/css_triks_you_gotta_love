		$('body').on('mousemove', '.button', (e) => {
			let x = e.pageX - e.target.offsetLeft;
			let y = e.pageY - e.target.offsetTop;
			$(e.target).css({'--x': x+'px', '--y': y+'px' });
		});

Query block width with calc()


		.block {
			display: inline-block;
			min-width: 50%;
			max-width: 100%;
			width: calc((600px - 100%) * 600);
		}
	


		/* When parent is 500px wide... */
		
		.block {
			min-width: 50%;   /* 250px WINNER */
			max-width: 100%;  /* 500px */
			width: calc(480px - 100%) * 480); /* -9600px */
		}
	


		/* When parent is 400px wide... */
		.block {
			min-width: 50%;  /* 200px */
			max-width: 100%; /* 400px WINNER */
			width: calc(480px - 100%) * 480); /* 38400px */
		}
